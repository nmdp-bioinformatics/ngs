023package org.nmdp.ngs.variant.vcf;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.util.ArrayList;
028import java.util.List;
029
030import javax.annotation.concurrent.Immutable;
031
032import com.google.common.collect.ImmutableList;
033
034/**
035 * VCF header.
036 */
037@Immutable
038public final class VcfHeader {
039    /** File format, e.g. <code>VCFv4.2</code>, the only required header field. */
040    private final String fileFormat;
041
042    /** List of meta-information header lines. */
043    private final List<String> meta;
044
045
046    /**
047     * Create a new VCF header.
048     *
049     * @param fileFormat file format, must not be null
050     * @param meta list of meta-information heder lines, must not be null
051     */
052    VcfHeader(final String fileFormat, final List<String> meta) {
053        checkNotNull(fileFormat);
054        checkNotNull(meta);
055        this.fileFormat = fileFormat;
056        this.meta = ImmutableList.copyOf(meta);
057    }
058
059
060    /**
061     * Return the file format for this VCF header.
062     *
063     * @return the file format for this VCF header
064     */
065    public String getFileFormat() {
066        return fileFormat;
067    }
068
069    /**
070     * Return the meta-information header lines for this VCF header.
071     *
072     * @return the meta-information header lines for this VCF header
073     */
074    public List<String> getMeta() {
075        return meta;
076    }
077
078
079    /**
080     * Create and return a new VCF header builder.
081     *
082     * @return a new VCF header builder
083     */
084    public static Builder builder() {
085        return new Builder();
086    }
087
088    /**
089     * VCF header builder.
090     */
091    public static final class Builder {
092        /** File format. */
093        private String fileFormat;
094
095        /** List of meta-information header lines. */
096        private final List<String> meta = new ArrayList<String>();
097
098
099        /**
100         * Private no-arg constructor.
101         */
102        private Builder() {
103            // empty
104        }
105
106
107        /**
108         * Return this VCF header builder configured with the specified file format.
109         *
110         * @param fileFormat file format
111         * @return this VCF header builder configured with the specified file format
112         */
113        public Builder withFileFormat(final String fileFormat) {
114            this.fileFormat = fileFormat;
115            return this;
116        }
117
118        /**
119         * Return this VCF header builder configured with the specified meta-information header line.
120         *
121         * @param meta meta-information header line, must not be null
122         * @return this VCF header builder configured with the specified meta-information header line
123         */
124        public Builder withMeta(final String meta) {
125            checkNotNull(meta);
126            this.meta.add(meta);
127            return this;
128        }
129
130        /**
131         * Return this VCF header builder configured with the specified meta-information header lines.
132         *
133         * @param meta one or more meta-information header lines, must not contain null
134         * @return this VCF header builder configured with the specified meta-information header line
135         */
136        public Builder withMeta(final String... meta) {
137            checkNotNull(meta);
138            for (String m : meta) {
139                withMeta(m);
140            }
141            return this;
142        }
143
144        /**
145         * Return this VCF header builder configured with the specified meta-information header lines.
146         *
147         * @param meta list of meta-information header lines, must not be null and must not contain null
148         * @return this VCF header builder configured with the specified meta-information header lines
149         */
150        public Builder withMeta(final List<String> meta) {
151            checkNotNull(meta);
152            for (String m : meta) {
153                withMeta(m);
154            }
155            return this;
156        }
157
158        /**
159         * Reset this VCF header builder.
160         *
161         * @return this VCF header builder
162         */
163        public Builder reset() {
164            fileFormat = null;
165            meta.clear();
166            return this;
167        }
168
169        /**
170         * Create and return a new VCF header populated from the configuration of this VCF header builder.
171         *
172         * @return a new VCF header populated from the configuration of this VCF header builder
173         */
174        public VcfHeader build() {
175            // todo: copy fileFormat to first line of meta?
176            return new VcfHeader(fileFormat, meta);
177        }
178    }
179}
