070public final class ValidateInterpretation implements Callable<Integer> {
071    private final File observedFile;
072    private final File outputFile;
073    private final File expectedFile;
074    private final int resolution;
075    private final List<String> loci;
076    private final boolean printSummary;
077    private final GlClient glclient;
078    static final int DEFAULT_RESOLUTION = 2;
079    static final List<String> DEFAULT_LOCI = ImmutableList.of("HLA-A", "HLA-B", "HLA-C", "HLA-DRB1", "HLA-DQB1");
080    private static final String USAGE = "ngs-validate-interpretation -e expected.txt -b observed.txt -r 2 -l \"HLA-A,HLA-B\"";

083    /**
084     * Validate interpretation.
085     *
086     * @param expectedFile expected file, at least one of expected or observed file must not be null
087     * @param observedFile observed file, at least one of expected or observed file must not be null
088     * @param outputFile output file, if any
089     * @param resolution resolution, must be in the range [1..4]
090     * @param loci list of loci to validate, must not be null
091     * @param printSummary print summary report
092     * @param glclient genotype list client, must not be null
093     */
094    public ValidateInterpretation(final File expectedFile, final File observedFile, final File outputFile, final int resolution, final List<String> loci, final boolean printSummary, final GlClient glclient) {
095        checkNotNull(loci);
096        checkNotNull(glclient);
097        checkArgument(expectedFile != null || observedFile != null, "at least one of expected or observed file must not be null");
098        checkArgument(resolution > 0 && resolution < 5, "resolution must be in the range [1..4]");
099        this.observedFile = observedFile;
100        this.expectedFile = expectedFile;
101        this.outputFile = outputFile;
102        this.resolution = resolution;
103        this.loci = loci;
104        this.printSummary = printSummary;
105        this.glclient = glclient;
106    }
107
108
109    @Override
110    public Integer call() throws Exception {
111        PrintWriter writer = null;
112
113        int passes = 0;
114        int failures = 0;
115        try {
116            writer = writer(outputFile);
117
118            // interpretations keyed by sample
119            ListMultimap<String, Interpretation> expected = read(expectedFile);
120            ListMultimap<String, Interpretation> observed = read(observedFile);
121
122            // for each sample in observed
123            for (String sample : observed.keySet()) {
124                for (Interpretation o : observed.get(sample)) {
125                    AlleleList observedAlleleList = asAlleleList(o);
126                    if (shouldValidate(o, observedAlleleList)) {
127
128                        // for each matching sample in expected
129                        for (Interpretation e : expected.get(sample)) {
130                            Genotype expectedGenotype = asGenotype(e);
131                            for (Haplotype expectedHaplotype : expectedGenotype.getHaplotypes()) {
132                                for (AlleleList expectedAlleleList : expectedHaplotype.getAlleleLists()) {
133                                    if (sameLocus(observedAlleleList, expectedAlleleList)) {
134
135                                        // for each pair of expected and observed alleles
136                                        for (Allele expectedAllele : expectedAlleleList.getAlleles()) {
137                                            boolean match = false;
138
139                                            for (Allele observedAllele : observedAlleleList.getAlleles()) {
140                                                if (matchByField(expectedAllele.getGlstring(), observedAllele.getGlstring()) >= resolution) {
141                                                    match = true;
142                                                    break;
143                                                }
144                                            }
145
146                                            if (match) {
147                                                passes++;
148                                            }
149                                            else {
150                                                failures++;
151                                            }
152
153                                            if (!printSummary) {
154                                                writer.println((match ? "PASS" : "FAIL") + "\t" + sample + "\t" + expectedAllele);
155                                            }
156                                        }
157                                    }
158                                }
159                            }
160                        }
161                    }
162                }
163            }
164            if (printSummary) {
165                writer.println("PASS\t" + passes);
166                writer.println("FAIL\t" + failures);
167            }
168
169            return 0;
170        }
171        finally {
172            try {
173                writer.close();
174            }
175            catch (Exception e) {
176                e.printStackTrace();
177                System.exit(1);
178            }
179        }
180    }

182    boolean shouldValidate(final Interpretation interpretation, final AlleleList alleleList) {
183        checkNotNull(interpretation);
184        checkNotNull(alleleList);
185        return loci.contains(interpretation.locus()) && interpretation.locus().equals(alleleList.getAlleles().get(0).getLocus().getGlstring());
186    }

201    AlleleList asAlleleList(final Interpretation interpretation) throws IOException {
202        checkNotNull(interpretation);
203        try {
204            return glclient.createAlleleList(interpretation.glstring());
205        }
206        catch (GlClientException e) {
207            throw new IOException("could not convert interpretation to allele list, caught " + e.getMessage(), e);
208        }
209    }
210
211    Genotype asGenotype(final Interpretation interpretation) throws IOException {
212        checkNotNull(interpretation);
213        try {
214            return glclient.createGenotype(interpretation.glstring());
215        }
216        catch (GlClientException e) {
217            throw new IOException("could not convert interpretation to genotype, caught " + e.getMessage(), e);
218        }
219    }

222    static int matchByField(final String allele0, final String allele1) {
223        checkNotNull(allele0);
224        checkNotNull(allele1);
225        List<String> allele0Parts = Splitter.on(":").splitToList(allele0);
226        List<String> allele1Parts = Splitter.on(":").splitToList(allele1);
227        int smallest = Math.min(allele0Parts.size(), allele1Parts.size());
228
229        for (int i = 0; i < smallest; i++) {
230            if (!allele0Parts.get(i).equals(allele1Parts.get(i))) {
231                return i;
232            }
233        }
234        return smallest;
235    }
236
237    static boolean sameLocus(final AlleleList alleleList0, final AlleleList alleleList1) {
238        checkNotNull(alleleList0);
239        checkNotNull(alleleList1);
240        return (alleleList0.getAlleles().get(0).getLocus().equals(alleleList1.getAlleles().get(0).getLocus()));
241    }

243    static ListMultimap<String, Interpretation> read(final File file) throws IOException {
244        BufferedReader reader = null;
245        ListMultimap<String, Interpretation> interpretations = ArrayListMultimap.create();
246        Interpretation.Builder builder = Interpretation.builder();
247        try {
248            reader = reader(file);
249            while (reader.ready()) {
250                String line = reader.readLine();
251                if (line == null) {
252                    break;
253                }
254                String[] tokens = line.split("\t");
255                if (tokens.length < 6) {
256                    throw new IOException("illegal interpretation format, expected at least 6 columns, found " + tokens.length);
257                }
258                Interpretation interpretation = builder.reset()
259                    .withSample(tokens[0])
260                    .withLocus(tokens[1])
261                    .withGeneFamily(tokens[2])
262                    .withAlleleDb(tokens[3])
263                    .withAlleleVersion(tokens[4])
264                    .withGlstring(tokens[5])
265                    .withConsensus(tokens.length > 6 ? tokens[6] : null)
266                    .build();
267
268                interpretations.put(interpretation.sample(), interpretation);
269            }
270            return interpretations;
271        }
272        finally {
273            try {
274                reader.close();
275            }
276            catch (Exception e) {
277                // ignore
278            }
279        }
280    }

282    static final class Interpretation {
283        private final String sample;
284        private final String locus;
285        private final String geneFamily;
286        private final String alleleDb;
287        private final String alleleVersion;
288        private final String glstring;
289        private final String consensus;
290
291        private Interpretation(final String sample,
292                               final String locus,
293                               final String geneFamily,
294                               final String alleleDb,
295                               final String alleleVersion,
296                               final String glstring,
297                               final String consensus) {
298            this.sample = sample;
299            this.locus = locus;
300            this.geneFamily = geneFamily;
301            this.alleleDb = alleleDb;
302            this.alleleVersion = alleleVersion;
303            this.glstring = glstring;
304            this.consensus = consensus;
305        }
306
307        String sample() {
308            return sample;
309        }
310
311        String locus() {
312            return locus;
313        }
314
315        String geneFamily() {
316            return geneFamily;
317        }
318
319        String alleleDb() {
320            return alleleDb;
321        }
322
323        String alleleVersion() {
324            return alleleVersion;
325        }
326
327        String glstring() {
328            return glstring;
329        }
330
331        String consensus() {
332            return consensus;
333        }
334
335        static Builder builder() {
336            return new Builder();
337        }
338
339        static final class Builder {
340            private String sample;
341            private String locus;
342            private String geneFamily;
343            private String alleleDb;
344            private String alleleVersion;
345            private String glstring;
346            private String consensus;
347
348            private Builder() {
349                // empty
350            }
351
352            Builder withSample(final String sample) {
353                this.sample = sample;
354                return this;
355            }
356
357            Builder withLocus(final String locus) {
358                this.locus = locus;
359                return this;
360            }
361
362            Builder withGeneFamily(final String geneFamily) {
363                this.geneFamily = geneFamily;
364                return this;
365            }
366
367            Builder withAlleleDb(final String alleleDb) {
368                this.alleleDb = alleleDb;
369                return this;
370            }
371
372            Builder withAlleleVersion(final String alleleVersion) {
373                this.alleleVersion = alleleVersion;
374                return this;
375            }
376
377            Builder withGlstring(final String glstring) {
378                this.glstring = glstring;
379                return this;
380            }
381
382            Builder withConsensus(final String consensus) {
383                this.consensus = consensus;
384                return this;
385            }
386
387            Builder reset() {
388                sample = null;
389                locus = null;
390                geneFamily = null;
391                alleleDb = null;
392                alleleVersion = null;
393                glstring = null;
394                consensus = null;
395                return this;
396            }
397
398            Interpretation build() {
399                return new Interpretation(sample, locus, geneFamily, alleleDb, alleleVersion, glstring, consensus);
400            }
401        }

405    /**
406     * Main.
407     *
408     * @param args command line args
409     */
410    public static void main(final String[] args) {
411        Switch about = new Switch("a", "about", "display about message");
412        Switch help = new Switch("h", "help", "display help message");
413
414        FileArgument expectedFile = new FileArgument("e", "expected-file", "expected interpretation file, default stdin; at least one of expected or observed file must be provided", false);
415        FileArgument observedFile = new FileArgument("b", "observed-file", "observed interpretation file, default stdin; at least one of expected or observed file must be provided", false);
416        FileArgument outputFile = new FileArgument("o", "output-file", "output file, default stdout", false);
417        IntegerArgument resolution = new IntegerArgument("r", "resolution", "resolution, must be in the range [1..4], default " + DEFAULT_RESOLUTION, false);
418        StringListArgument loci = new StringListArgument("l", "loci", "list of loci to validate, default " + DEFAULT_LOCI, false);
419        Switch printSummary = new Switch("s", "summary", "print summary");
420
421        ArgumentList arguments = new ArgumentList(about, help, expectedFile, observedFile, outputFile, resolution, loci, printSummary);
422        CommandLine commandLine = new CommandLine(args);
423
424        ValidateInterpretation validateInterpretation = null;
425        try
426        {
427            CommandLineParser.parse(commandLine, arguments);
428            if (about.wasFound()) {
429                About.about(System.out);
430                System.exit(0);
431            }
432            if (help.wasFound()) {
433                Usage.usage(USAGE, null, commandLine, arguments, System.out);
434                System.exit(0);
435            }
436
437            // todo: allow for configuration of glclient
438            validateInterpretation = new ValidateInterpretation(expectedFile.getValue(), observedFile.getValue(), outputFile.getValue(), resolution.getValue(DEFAULT_RESOLUTION), loci.getValue(DEFAULT_LOCI), printSummary.wasFound(), LocalGlClient.create());
439        }
440        catch (CommandLineParseException e) {
441            if (about.wasFound()) {
442                About.about(System.out);
443                System.exit(0);
444            }
445            if (help.wasFound()) {
446                Usage.usage(USAGE, null, commandLine, arguments, System.out);
447                System.exit(0);
448            }
449            Usage.usage(USAGE, e, commandLine, arguments, System.err);
450            System.exit(-1);
451        }
452        try {
453            System.exit(validateInterpretation.call());
454        }
455        catch (Exception e) {
456            e.printStackTrace();
457            System.exit(1);
458        }
459    }
