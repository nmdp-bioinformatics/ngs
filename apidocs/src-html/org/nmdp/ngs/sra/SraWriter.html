070    public static void writeAnalysis(final Analysis analysis, final Writer writer) throws IOException {
071        checkNotNull(analysis);
072        checkNotNull(writer);
073
074        try {
075            JAXBContext context = JAXBContext.newInstance(Analysis.class);
076            Marshaller marshaller = context.createMarshaller();
077            SchemaFactory schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
078            URL commonSchemaURL = SraReader.class.getResource("/org/nmdp/ngs/sra/xsd/SRA.common.xsd");
079            URL analysisSchemaURL = SraReader.class.getResource("/org/nmdp/ngs/sra/xsd/SRA.analysis.xsd");
080            Schema schema = schemaFactory.newSchema(new StreamSource[] { new StreamSource(commonSchemaURL.toString()), new StreamSource(analysisSchemaURL.toString()) });
081            marshaller.setSchema(schema);
082            marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
083            marshaller.marshal(analysis, writer);
084        }
085        catch (JAXBException | SAXException e) {
086            throw new IOException("could not marshal Analysis", e);
087        }
088    }
089
090    public static void writeAnalysis(final Analysis analysis, final File file) throws IOException {
091        checkNotNull(analysis);
092        checkNotNull(file);
093        try (BufferedWriter writer = new BufferedWriter(new FileWriter(file))) {
094            writeAnalysis(analysis, writer);
095        }
096    }
