023package org.nmdp.ngs.sra;
024
025import java.util.Map;
026
027import com.google.common.collect.ImmutableMap;
028import com.google.common.collect.Maps;
029
030/**
031 * Instrument model.
032 */
033public enum InstrumentModel {
034    ILLUMINA_GENOME_ANALYZER("Illumina Genome Analyzer"),
035    ILLUMINA_GENOME_ANALYZER_II("Illumina Genome Analyzer II"),
036    ILLUMINA_GENOME_ANALYZER_IIX("Illumina Genome Analyzer IIx"),
037    ILLUMINA_HISEQ_2500("Illumina HiSeq 2500"),
038    ILLUMINA_HISEQ_2000("Illumina HiSeq 2000"),
039    ILLUMINA_HISEQ_1000("Illumina HiSeq 1000"),
040    ILLUMINA_MISEQ("Illumina MiSeq"),
041    ILLUMINA_HISCANSQ("Illumina HiScanSQ"),
042    ROCHE_454_GS("454 GS"),
043    ROCHE_454_GS_20("454 GS 20"),
044    ROCHE_454_GS_FLX("454 GS FLX"),
045    ROCHE_454_GS_FLX_PLUS("454 GS FLX+"),
046    ROCHE_454_GS_FLX_TITANIUM("454 GS FLX Titanium"),
047    ROCHE_454_GS_JUNIOR("454 GS Junior"),
048    AB_SOLID_SYSTEM("AB SOLiD System"),
049    AB_SOLID_SYSTEM_2_0("AB SOLiD System 2.0"),
050    AB_SOLID_SYSTEM_3_0("AB SOLiD System 3.0"),
051    AB_SOLID_3_PLUS_SYSTEM("AB SOLiD 3 Plus System"),
052    AB_SOLID_4_SYSTEM("AB SOLiD 4 System"),
053    AB_SOLID_4HQ_SYSTEM("AB SOLiD 4hq System"),
054    AB_SOLID_PI_SYSTEM("AB SOLiD PI System"),
055    AB_5500_GENETIC_ANALYZER("AB 5500 Genetic Analyzer"),
056    AB_5500XL_GENETIC_ANALYZER("AB 5500xl Genetic Analyzer"),
057    UNSPECIFIED("unspecified");
058
059    private final String description;
060    private static final Map<String, InstrumentModel> BY_DESCRIPTION;
061
062    static
063    {
064        Map<String, InstrumentModel> byDescription = Maps.newHashMap();
065        for (InstrumentModel studyType : values()) {
066            byDescription.put(studyType.getDescription(), studyType);
067        }
068        BY_DESCRIPTION = ImmutableMap.copyOf(byDescription);
069    }
070
071
072    /**
073     * Create a new instrument model with the specified description.
074     *
075     * @param description description
076     */
077    private InstrumentModel(final String description) {
078        this.description = description;
079    }
080
081
082    /**
083     * Return the description of this instrument model.
084     *
085     * @return the description of this instrument model
086     */
087    public String getDescription() {
088        return description;
089    }
090
091    /**
092     * Return a map of instrument models keyed by description.
093     *
094     * @return a map of instrument models keyed by description
095     */
096    public static Map<String, InstrumentModel> byDescription() {
097        return BY_DESCRIPTION;
098    }
099
100    /**
101     * Return the instrument model with the specified description, or null if no such instrument model exists.
102     *
103     * @param description description
104     * @return the instrument model with the specified description, or null if no such instrument model exists
105     */
106    public static InstrumentModel fromDescription(final String description) {
107        return BY_DESCRIPTION.get(description);
108    }
109}
