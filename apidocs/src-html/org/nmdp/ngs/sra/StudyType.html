023package org.nmdp.ngs.sra;
024
025import java.util.Map;
026
027import com.google.common.collect.ImmutableMap;
028import com.google.common.collect.Maps;
029
030/**
031 * Study type.
032 */
033public enum StudyType {
034    WHOLE_GENOME_SEQUENCING("Whole Genome Sequencing"),
035    METAGENOMICS("Metagenomics"),
036    TRANSCRIPTOME_ANALYSIS("Transcriptome Analysis"),
037    RESEQUENCING("Resequencing"),
038    EPIGENETICS("Epigenetics"),
039    SYNTHETIC_GENOMICS("Synthetic Genomics"),
040    FORENSIC_OR_PALEO_GENOMICS("Forensic or Paleo-genomics"),
041    GENE_REGULATION_STUDY("Gene Regulation Study"),
042    CANCER_GENOMICS("Cancer Genomics"),
043    POPULATION_GENOMICS("Population Genomics"),
044    RNA_SEQ("RNASeq"),
045    EXOME_SEQUENCING("Exome Sequencing"),
046    POOLED_CLONE_SEQUENCING("Pooled Clone Sequencing"),
047    OTHER("Other");
048
049    private final String description;
050    private static final Map<String, StudyType> BY_DESCRIPTION;
051
052    static
053    {
054        Map<String, StudyType> byDescription = Maps.newHashMap();
055        for (StudyType studyType : values()) {
056            byDescription.put(studyType.getDescription(), studyType);
057        }
058        BY_DESCRIPTION = ImmutableMap.copyOf(byDescription);
059    }
060
061    /**
062     * Create a new study type with the specified description.
063     *
064     * @param description description
065     */
066    private StudyType(final String description) {
067        this.description = description;
068    }
069
070
071    /**
072     * Return the description of this study type.
073     *
074     * @return the description of this study type
075     */
076    public String getDescription() {
077        return description;
078    }
079
080    /**
081     * Return a map of study types keyed by description.
082     *
083     * @return a map of study types keyed by description
084     */
085    public static Map<String, StudyType> byDescription() {
086        return BY_DESCRIPTION;
087    }
088
089    /**
090     * Return the study type with the specified description, or null if no such study type exists.
091     *
092     * @param description description
093     * @return the study type with the specified description, or null if no such study type exists
094     */
095    public static StudyType fromDescription(final String description) {
096        return BY_DESCRIPTION.get(description);
097    }
098}
