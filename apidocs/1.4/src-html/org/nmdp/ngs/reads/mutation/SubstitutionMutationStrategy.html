023package org.nmdp.ngs.reads.mutation;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.util.List;
028
029import com.google.common.collect.ImmutableList;
030
031import org.apache.commons.math3.random.RandomGenerator;
032
033import org.biojava.bio.seq.DNATools;
034
035import org.biojava.bio.symbol.AtomicSymbol;
036import org.biojava.bio.symbol.IllegalSymbolException;
037import org.biojava.bio.symbol.SimpleSymbolList;
038import org.biojava.bio.symbol.Symbol;
039import org.biojava.bio.symbol.SymbolList;
040
041import org.nmdp.ngs.reads.MutationStrategy;
042
043/**
044 * Substitution mutation strategy.
045 */
046public final class SubstitutionMutationStrategy implements MutationStrategy {
047    /** Random. */
048    private final RandomGenerator random;
049
050    /** List of DNA symbols. */
051    private static final List<AtomicSymbol> SYMBOLS = ImmutableList.of(DNATools.a(), DNATools.c(), DNATools.g(), DNATools.t());
052
053
054    /**
055     * Create a new substitution mutation strategy.
056     *
057     * @param random random, must not be null
058     */
059    public SubstitutionMutationStrategy(final RandomGenerator random) {
060        checkNotNull(random);
061        this.random = random;
062    }
063
064
065    @Override
066    public SymbolList mutate(final Symbol symbol) {
067        SimpleSymbolList symbolList = new SimpleSymbolList(DNATools.getDNA());
068        int index = random.nextInt(SYMBOLS.size());
069        try {
070            symbolList.addSymbol(SYMBOLS.get(index));
071        }
072        catch (IllegalSymbolException e) {
073            // ignore
074        }
075        return symbolList;
076    }
077}
