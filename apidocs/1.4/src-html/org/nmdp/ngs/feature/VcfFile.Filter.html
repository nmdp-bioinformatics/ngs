056    public final class Iterator implements java.util.Iterator<VcfRecord> {
057        private long lineNumber;
058        private final InputStream stream;
059        private final byte [] buffer = new byte[BUFFER_SIZE];
060        public int bytes;
061
062        public Iterator(final InputStream stream) {
063            lineNumber = 0;
064            bytes = 0;
065            this.stream = stream;
066        }
067
068        @Override
069        public boolean hasNext() {
070            try {
071                return stream.available() == 1;
072            }
073            catch (IOException exception) {
074                return false;
075            }
076        }
077
078        @Override
079        public VcfRecord next() {
080            try {
081                bytes += stream.read(buffer);
082                return FeatureParser.parseVcfRecord(Arrays.toString(buffer), lineNumber);
083            }
084            catch (IOException | ParseException exception) {
085                return null;
086            }
087        }
088
089        @Override
090        public void remove() {
091            // empty
092        }
093    }
094
095    public Iterator iterator() throws IOException {
096        stream.reset();
097        return new Iterator(stream);
098    }

133    public static class Metadata {
134        public final static String ID = "ID";
135        public final static String DESCRIPTION = "Description";
136        public final static String INTEGER ="Integer";
137        public final static String FLOAT = "Float";
138        public final static String CHARACTER = "Character";
139        public final static String STRING = "String";
140        public final static String FLAG = "Flag";
141
142        public enum Type {
143            UNKNOWN,
144            INTEGER,
145            FLOAT,
146            CHARACTER,
147            STRING,
148            FLAG
149        }
150
151        private String name;
152        private String description;
153        protected SimpleAnnotation annotation;
154
155        public Metadata(final String name, final String description) {
156            this.name = name;
157            this.description = description;
158        }
159    
160        public Metadata(final SimpleAnnotation annotation) throws VcfFileException {
161            this.annotation = annotation;
162            try {
163                name = ((ArrayList<String>) this.annotation.getProperty(ID)).get(0);
164            }
165            catch(NoSuchElementException exception) {
166                throw new VcfFileException("missing metadata field \"" + ID + "\"");
167            }
168
169            try {
170                description = ((ArrayList<String>) this.annotation.getProperty(DESCRIPTION)).get(0);
171            }
172            catch(NoSuchElementException exception) {
173                throw new VcfFileException("missing metadata field \"" + DESCRIPTION + "\"");
174            }
175        }
176
177        public final String getName() {
178            return name;
179        }
180
181        public final String getDescription() {
182            return description.replace(">", "");
183        }
184
185        public final SimpleAnnotation getFullAnnotation() {
186            return annotation;
187        }
188    }

190    public static class Info extends VcfFile.Metadata {
191        public final static String NUMBER = "Number";
192        public final static String TYPE = "Type";
193
194        private Type type;
195        private Object number;
196
197        public Info(final SimpleAnnotation annotation) throws VcfFileException {
198            super(annotation);
199
200            try {
201                number = ((ArrayList<String>) this.annotation.getProperty(NUMBER)).get(0);
202            }
203            catch(NoSuchElementException exception) {
204                throw new VcfFileException("missing " + INFO + " field \"" + NUMBER + "\"");
205            }
206
207            try {
208                String argument = ((ArrayList<String>) this.annotation.getProperty(TYPE)).get(0);
209
210                switch (argument) {
211                case INTEGER:
212                    type = VcfFile.Metadata.Type.INTEGER;
213                    break;
214                case FLOAT:
215                    type = VcfFile.Metadata.Type.FLOAT;
216                    break;
217                case STRING:
218                    type = VcfFile.Metadata.Type.STRING;
219                    break;
220                case FLAG:
221                    type = VcfFile.Metadata.Type.FLAG;
222                    break;
223                case CHARACTER:
224                    type = VcfFile.Metadata.Type.CHARACTER;
225                    break;
226                default:
227                    type = VcfFile.Metadata.Type.UNKNOWN;
228                    break;
229                }
230            }
231            catch(NoSuchElementException exception) {
232                throw new VcfFileException("missing " + INFO + " field \"" + TYPE + "\"");
233            }
234        }
235
236        public Object getNumber() {
237            return number;
238        }
239
240        public Type getType() {
241            return type;
242        }
243    }

257    public VcfFile(final String filename) throws IOException, ParseException {
258        this.filename = filename;
259        this.format = "UNKNOWN";
260        this.reference = "UNKNOWN";
261        this.header = null;
262        this.metadata = new ArrayList<Metadata>();
263        strictness = Strictness.NONE;
264        bytes = 0;
265
266        this.stream = new FileInputStream(filename);
267
268        if(filename.endsWith(".gz")) {
269            try {
270                this.stream = new GZIPInputStream(stream);
271            }
272            finally {
273                // stream.close();
274            }
275        }
276
277        bytes += stream.read(buffer);
278        String line = new String(buffer, "UTF8");
279        //System.out.println("bytes = " + bytes + " line = " + line);
280
281        while(line.startsWith("#")) {
282            metadata.add(FeatureParser.parseVcfMetadata(line));
283            bytes += stream.read(buffer);
284            line = Arrays.toString(buffer);
285            //System.out.println("bytes = " + bytes + " line = " + line);
286        }
287
288        stream.mark(120);
289    }
